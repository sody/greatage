7   import java.lang.reflect.Constructor;

9   /**
10   * This class represents abstract implementation of all types of service configuration. Adds auto build functionality
11   * for configuration items instantiation.
12   *
13   * @param <T> service type
14   * @param <V> type of configuration items
15   * @param <C> configuration resulting type
16   * @author Ivan Khalopik
17   * @since 1.0
18   */
19  public abstract class AbstractConfiguration<T, V, C> {
20  	private final ServiceResources<T> resources;
21  
22  	/**
23  	 * Creates new instance of service configuration with defined service resources.
24  	 *
25  	 * @param resources service resources
26  	 */
27  	protected AbstractConfiguration(final ServiceResources<T> resources) {
28  		this.resources = resources;
29  	}
30  
31  	/**
32  	 * Automatically builds new instance of service configuration item by specified item class.
33  	 *
34  	 * @param valueClass service configuration item class
35  	 * @return automatically built new instance of service configuration item
36  	 */
37  	protected V newInstance(final Class<? extends V> valueClass) {
38  		try {
39  			final Constructor constructor = valueClass.getConstructors()[0];
40  			final Object[] parameters = InternalUtils.calculateParameters(resources, constructor);
41  			return valueClass.cast(constructor.newInstance(parameters));
42  		} catch (Exception e) {
43  			throw new RuntimeException(String.format("Can't create object of class '%s'", valueClass), e);
44  		}
45  	}
46  
47  	/**
48  	 * Builds resulting service configuration. Used for calculating configuration instance for service construction
49  	 * method.
50  	 *
51  	 * @return resulting service configuration
52  	 */
53  	protected abstract C build();
54  }
