5   package org.greatage.ioc;

7   import org.greatage.ioc.proxy.MethodAdvice;
8   import org.greatage.ioc.proxy.ObjectBuilder;
9   import org.greatage.ioc.proxy.ProxyFactory;
10  import org.greatage.util.DescriptionBuilder;
11  import org.greatage.util.Locker;
12  import org.greatage.util.OrderingUtils;
13  
14  import java.util.ArrayList;
15  import java.util.List;
16  
17  /**
18   * @author Ivan Khalopik
19   * @since 1.0
20   */
21  public class ScopedService<T> implements ServiceStatus<T> {
22  	private final ServiceResources<T> resources;
23  	private final ObjectBuilder<T> builder;
24  	private final List<Interceptor<T>> interceptors;
25  
26  	private final Locker locker = new Locker();
27  
28  	private T serviceInstance;
29  
30  	ScopedService(final ServiceLocator locator,
31  				  final Service<T> service,
32  				  final List<Contributor<T>> contributors,
33  				  final List<Decorator<T>> decorators,
34  				  final List<Interceptor<T>> interceptors) {
35  		this.resources = new ServiceInitialResources<T>(locator, service);
36  		final ServiceBuilder<T> serviceBuilder = new ServiceBuilder<T>(resources, service, contributors, decorators);
37  		this.builder = new ScopedBuilder<T>(resources, serviceBuilder);
38  		this.interceptors = interceptors;
39  	}
40  
41  	public String getServiceId() {
42  		return resources.getServiceId();
43  	}
44  
45  	public Class<T> getServiceClass() {
46  		return resources.getServiceClass();
47  	}
48  
49  	public String getServiceScope() {
50  		return resources.getServiceScope();
51  	}
52  
53  	public T getService() {
54  		if (serviceInstance == null) {
55  			locker.lock();
56  			final ProxyFactory proxyFactory = resources.getResource(ProxyFactory.class);
57  			serviceInstance = proxyFactory.createProxy(builder, createAdvices());
58  		}
59  		return serviceInstance;
60  	}
61  
62  	private List<MethodAdvice> createAdvices() {
63  		final List<MethodAdvice> advices = new ArrayList<MethodAdvice>();
64  		final List<Interceptor<T>> ordered = OrderingUtils.order(interceptors);
65  		for (Interceptor<T> interceptor : ordered) {
66  			final MethodAdvice advice = interceptor.intercept(resources);
67  			advices.add(advice);
68  		}
69  		return advices;
70  	}
71  
72  	@Override
73  	public String toString() {
74  		final DescriptionBuilder db = new DescriptionBuilder(getClass());
75  		db.append("resources", resources);
76  		db.append("builder", builder);
77  		return db.toString();
78  	}
79  }
