7   import org.greatage.util.DescriptionBuilder;
8   
9   import java.lang.reflect.Method;
10  import java.util.List;
11  
12  /**
13   * This class represents utility for lazy creation of object from object builder.
14   *
15   * @param <T> delegate type
16   * @author Ivan Khalopik
17   * @since 1.0
18   */
19  public abstract class AbstractInvocationHandler<T> {
20  	private final ObjectBuilder<T> builder;
21  	private final List<MethodAdvice> advices;
22  
23  	/**
24  	 * Creates new instance of utility for lazy creation of object from specified object builder.
25  	 *
26  	 * @param builder object builder
27  	 * @param advices method advices
28  	 */
29  	protected AbstractInvocationHandler(final ObjectBuilder<T> builder, final List<MethodAdvice> advices) {
30  		assert builder != null;
31  
32  		this.builder = builder;
33  		this.advices = advices;
34  	}
35  
36  	/**
37  	 * Gets object instance. If instance is not initialized, creates it from object builder.
38  	 *
39  	 * @return lazy initialized object instance
40  	 */
41  	protected T getDelegate() {
42  		return builder.build();
43  	}
44  
45  	/**
46  	 * Invokes specified method with specified parameters on delegate instance
47  	 *
48  	 * @param method	 interface method
49  	 * @param parameters invocation parameters
50  	 * @return delegated from invocation return object
51  	 * @throws Throwable if some problems occurs while invoking method
52  	 */
53  	protected Object invoke(final Method method, final Object... parameters) throws Throwable {
54  		final Method realMethod = getDelegate().getClass().getMethod(method.getName(), method.getParameterTypes());
55  		return createInvocation(realMethod).proceed(parameters);
56  	}
57  
58  	/**
59  	 * Creates new invocation instance for specified method with defined method advices.
60  	 *
61  	 * @param method method
62  	 * @return new invocation instance for specified method with defined method advices
63  	 */
64  	private Invocation createInvocation(final Method method) {
65  		Invocation invocation = new MethodInvocation(getDelegate(), method);
66  		if (advices != null) {
67  			for (MethodAdvice advice : advices) {
68  				invocation = new AdvisedInvocation(invocation, advice);
69  			}
70  		}
71  		return invocation;
72  	}
73  
74  	/**
75  	 * {@inheritDoc}
76  	 */
77  	@Override
78  	public String toString() {
79  		final DescriptionBuilder db = new DescriptionBuilder(getClass());
80  		db.append("builder", builder);
81  		db.append("advices", advices);
82  		return db.toString();
83  	}
