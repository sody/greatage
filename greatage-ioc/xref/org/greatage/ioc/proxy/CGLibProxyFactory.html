12   * This class represents proxy factory implementation using CGLib library.

17  public class CGLibProxyFactory extends AbstractProxyFactory {
18  
19  	public <T> T createProxy(final ObjectBuilder<T> objectBuilder, final List<MethodAdvice> advices) {
20  		validate(objectBuilder);
21  
22  		final Class superClass = objectBuilder.getObjectClass().isInterface() ? Object.class : objectBuilder.getObjectClass();
23  		final CGLibInvocationHandler<T> handler = new CGLibInvocationHandler<T>(objectBuilder, advices);
24  		final Object proxy = objectBuilder.getObjectClass().isInterface() ?
25  				Enhancer.create(superClass, new Class[]{objectBuilder.getObjectClass()}, handler) :
26  				Enhancer.create(superClass, handler);
27  		return objectBuilder.getObjectClass().cast(proxy);
28  	}
