5   package org.greatage.ioc;

7   import java.lang.annotation.Annotation;
8   import java.lang.reflect.Constructor;
9   import java.lang.reflect.Method;
10  
11  /**
12   * This class represents internal utility methods for calculating service dependencies.
13   *
14   * @author Ivan Khalopik
15   * @since 1.0
16   */
17  public class InternalUtils {
18  
19  	/**
20  	 * Calculates service dependencies according to specified build constructor.
21  	 *
22  	 * @param resources   service resources
23  	 * @param constructor service build constructor
24  	 * @param <T>         service type
25  	 * @return array of calculated service dependencies
26  	 */
27  	public static <T> Object[] calculateParameters(final ServiceResources<T> resources, final Constructor constructor) {
28  		return calculateInjections(resources, constructor.getParameterTypes(), constructor.getParameterAnnotations());
29  	}
30  
31  	/**
32  	 * Calculates service dependencies according to specified build method.
33  	 *
34  	 * @param resources service resources
35  	 * @param method	service build method
36  	 * @param <T>       service type
37  	 * @return array of calculated service dependencies
38  	 */
39  	public static <T> Object[] calculateParameters(final ServiceResources<T> resources, final Method method) {
40  		return calculateInjections(resources, method.getParameterTypes(), method.getParameterAnnotations());
41  	}
42  
43  	/**
44  	 * Calculates service dependencies according to specified dependency types and annotations.
45  	 *
46  	 * @param resources   service resources
47  	 * @param types	   dependency types
48  	 * @param annotations dependency annotations
49  	 * @param <T>         service type
50  	 * @return array of calculated service dependencies
51  	 */
52  	private static <T> Object[] calculateInjections(final ServiceResources<T> resources,
53  													final Class<?>[] types, final Annotation[][] annotations) {
54  		final int count = types.length;
55  		final Object[] parameters = new Object[count];
56  		for (int i = 0; i < count; i++) {
57  			parameters[i] = resources.getResource(types[i], annotations[i]);
58  		}
59  		return parameters;
60  	}
61  }
