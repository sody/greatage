005    package org.greatage.hibernate.type;
006    
007    import org.hibernate.MappingException;
008    import org.hibernate.usertype.ParameterizedType;
009    import org.hibernate.usertype.UserType;
010    
011    import java.io.Serializable;
012    import java.sql.PreparedStatement;
013    import java.sql.ResultSet;
014    import java.sql.SQLException;
015    import java.util.HashMap;
016    import java.util.Map;
017    import java.util.Properties;
018    
019    /**
020     * This class represents base class for all hibernate custom user types that represent enum as some simple sql type.
021     *
022     * @param <E> type of enum
023     * @param <V> type of corresponding value
024     * @author Ivan Khalopik
025     * @since 1.0
026     */
027    public abstract class AbstractEnumUserType<E extends Enum, V extends Serializable>
028                    implements UserType, ParameterizedType {
029            private final Map<V, E> constantsByValue = new HashMap<V, E>();
030            private final Map<E, V> constantsByEnum = new HashMap<E, V>();
031    
032            private Class<E> enumClass;
033            private final int sqlType;
034    
035            /**
036             * Constructor that creates hibernate custom user type that represent enum as some simple sql type.
037             *
038             * @param sqlType correspondent sql type
039             */
040            protected AbstractEnumUserType(final int sqlType) {
041                    this.sqlType = sqlType;
042            }
043    
044            /**
045             * Configure hibernate custom type with enum class name stored in <tt>enumClass</tt> parameter.
046             *
047             * @param parameters hibernate custom type configuration parameters
048             */
049            @SuppressWarnings("unchecked")
050            public void setParameterValues(final Properties parameters) {
051                    final String enumClassName = parameters.getProperty("enumClass");
052                    if (enumClassName == null) {
053                            throw new MappingException("enumClass parameter not specified");
054                    }
055    
056                    try {
057                            final Class<E> configuredEnumClass = (Class<E>) Class.forName(enumClassName);
058                            setEnumClass(configuredEnumClass);
059                    } catch (ClassNotFoundException e) {
060                            throw new MappingException("Enum class " + enumClassName + " not found", e);
061                    } catch (ClassCastException e) {
062                            throw new MappingException("Class " + enumClassName + " is not enum class", e);
063                    }
064            }
065    
066            /**
067             * Sets enum class and configures all enum-to-value and value-to-enum mappings.
068             *
069             * @param enumClass enum class
070             */
071            public void setEnumClass(final Class<E> enumClass) {
072                    this.enumClass = enumClass;
073                    constantsByValue.clear();
074                    for (E e : enumClass.getEnumConstants()) {
075                            final V value = enumToValue(e);
076                            constantsByValue.put(value, e);
077                            constantsByEnum.put(e, value);
078                    }
079            }
080    
081            /**
082             * Gets enum class correspondent to this type.
083             *
084             * @return enum class
085             */
086            public Class<E> getEnumClass() {
087                    return enumClass;
088            }
089    
090            /**
091             * Gets sql type correspondent to this type.
092             *
093             * @return sql type
094             */
095            public int getSqlType() {
096                    return sqlType;
097            }
098    
099            /**
100             * {@inheritDoc}
101             */
102            public Object nullSafeGet(final ResultSet rs, final String[] names, final Object owner) throws SQLException {
103                    final V value = get(rs, names[0]);
104                    return value == null || rs.wasNull() ? null : getEnum(value);
105            }
106    
107            /**
108             * {@inheritDoc}
109             */
110            @SuppressWarnings("unchecked")
111            public void nullSafeSet(final PreparedStatement st, final Object value, final int index) throws SQLException {
112                    if (value == null) {
113                            st.setNull(index, getSqlType());
114                    } else {
115                            final V v = getValue((E) value);
116                            set(st, v, index);
117                    }
118            }
119    
120            /**
121             * {@inheritDoc}
122             */
123            public Class<E> returnedClass() {
124                    return getEnumClass();
125            }
126    
127            /**
128             * {@inheritDoc}
129             */
130            public int[] sqlTypes() {
131                    return new int[]{sqlType};
132            }
133    
134            /**
135             * {@inheritDoc}
136             */
137            public Object deepCopy(final Object value) {
138                    return value;
139            }
140    
141            /**
142             * {@inheritDoc}
143             */
144            public Object replace(final Object original, final Object target, final Object owner) {
145                    return original;
146            }
147    
148            /**
149             * {@inheritDoc}
150             */
151            public boolean isMutable() {
152                    return false;
153            }
154    
155            /**
156             * {@inheritDoc}
157             */
158            public Serializable disassemble(final Object value) {
159                    return (Enum) value;
160            }
161    
162            /**
163             * {@inheritDoc}
164             */
165            public Object assemble(final Serializable cached, final Object owner) {
166                    return cached;
167            }
168    
169            /**
170             * {@inheritDoc}
171             */
172            public boolean equals(final Object x, final Object y) {
173                    return x == y;
174            }
175    
176            /**
177             * {@inheritDoc}
178             */
179            public int hashCode(final Object x) {
180                    return x.hashCode();
181            }
182    
183            /**
184             * Gets enum constant mapped to specified value.
185             *
186             * @param value enum value
187             * @return enum constant mapped to specified value
188             */
189            protected E getEnum(final V value) {
190                    return constantsByValue.get(value);
191            }
192    
193            /**
194             * Gets value correspondent to specified enum constant.
195             *
196             * @param enumConstant enum constant
197             * @return value correspondent to specified enum constant
198             */
199            protected V getValue(final E enumConstant) {
200                    return constantsByEnum.get(enumConstant);
201            }
202    
203            /**
204             * Gets value from result set for specified column name.
205             *
206             * @param rs   SQL result set
207             * @param name SQL column name
208             * @return value from result set for specified column name
209             * @throws SQLException if error occurs when retrieving value
210             */
211            protected abstract V get(final ResultSet rs, final String name) throws SQLException;
212    
213            /**
214             * Sets value to prepared statement for specified column index.
215             *
216             * @param st    SQL prepered statement
217             * @param index SQL column index
218             * @param value value that will be set to prepared statement
219             * @throws SQLException if error occurs when setting value
220             */
221            protected abstract void set(final PreparedStatement st, final V value, final int index) throws SQLException;
222    
223            /**
224             * Converts enum constant to value. It is used when initializing hibernate type purpose to generate mappings between
225             * enum constants and values of some simple type.
226             *
227             * @param enumConstant enum constant
228             * @return value correspondent to specified enum constant
229             */
230            protected abstract V enumToValue(final E enumConstant);
231    }
