005    package org.greatage.security.drools;
006    
007    import org.greatage.security.AccessControlEntry;
008    
009    /**
010     * @author Ivan Khalopik
011     * @since 1.0
012     */
013    public class DroolsAccessControlEntry implements AccessControlEntry {
014            private final String authority;
015            private final String permission;
016            private final Object object;
017    
018            private boolean granted;
019    
020            public DroolsAccessControlEntry(final Object object, final String authority, final String permission) {
021                    this.authority = authority;
022                    this.permission = permission;
023                    this.object = object;
024            }
025    
026            public String getAuthority() {
027                    return authority;
028            }
029    
030            public String getPermission() {
031                    return permission;
032            }
033    
034            public Object getSecuredObject() {
035                    return object;
036            }
037    
038            public boolean isGranted() {
039                    return granted;
040            }
041    
042            public void grant() {
043                    granted = true;
044            }
045    
046            public void deny() {
047                    granted = false;
048            }
049    
050    }
