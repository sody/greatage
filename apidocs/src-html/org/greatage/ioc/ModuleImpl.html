005    package org.greatage.ioc;
006    
007    import org.greatage.ioc.annotations.Bind;
008    import org.greatage.ioc.annotations.Build;
009    import org.greatage.ioc.annotations.Contribute;
010    import org.greatage.ioc.annotations.Decorate;
011    import org.greatage.ioc.annotations.Intercept;
012    import org.greatage.ioc.scope.ScopeConstants;
013    import org.greatage.util.CollectionUtils;
014    
015    import java.lang.reflect.Method;
016    import java.util.Collection;
017    import java.util.List;
018    
019    /**
020     * This class represents default implementation of module definition. It is based on automatically module creation by
021     * its class. It can also be represented as service definition with service identifier and class like module class.
022     *
023     * @param <T> module service type
024     * @author Ivan Khalopik
025     * @since 1.0
026     */
027    public class ModuleImpl<T> extends ServiceImpl<T> implements Module {
028            private final List<Service> services = CollectionUtils.newList();
029            private final List<Contributor> contributors = CollectionUtils.newList();
030            private final List<Decorator> decorators = CollectionUtils.newList();
031            private final List<Interceptor> interceptors = CollectionUtils.newList();
032    
033            /**
034             * Creates new instance of module definition for specified module class. It seeks for methods annotated with {@link
035             * Build}, {@link Contribute}, {@link Decorate}, {@link Intercept} and {@link Bind} annotations and creates for them
036             * service, contribute, decorate, intercept and bind definitions respectively.
037             *
038             * @param moduleClass module class
039             */
040            ModuleImpl(final Class<T> moduleClass) {
041                    super(moduleClass.getName(), moduleClass, ScopeConstants.GLOBAL, false);
042                    services.add(this);
043                    for (Method method : moduleClass.getMethods()) {
044                            if (method.isAnnotationPresent(Build.class)) {
045                                    services.add(new ServiceFactory(moduleClass, method));
046                            } else if (method.isAnnotationPresent(Contribute.class)) {
047                                    contributors.add(new ContributorImpl(moduleClass, method));
048                            } else if (method.isAnnotationPresent(Decorate.class)) {
049                                    decorators.add(new DecoratorImpl(moduleClass, method));
050                            } else if (method.isAnnotationPresent(Intercept.class)) {
051                                    interceptors.add(new InterceptorImpl(moduleClass, method));
052                            } else if (method.isAnnotationPresent(Bind.class)) {
053                                    final ServiceBinderImpl binder = new ServiceBinderImpl();
054                                    try {
055                                            method.invoke(null, binder);
056                                            services.addAll(binder.getServices());
057                                    } catch (Exception e) {
058                                            throw new RuntimeException("Exception in bind method", e);
059                                    }
060                            }
061                    }
062            }
063    
064            /**
065             * {@inheritDoc}
066             */
067            public Collection<Service> getServices() {
068                    return services;
069            }
070    
071            /**
072             * {@inheritDoc}
073             */
074            public <T> List<Contributor<T>> getContributors(final Service<T> service) {
075                    final List<Contributor<T>> result = CollectionUtils.newList();
076                    for (Contributor contributor : contributors) {
077                            if (contributor.supports(service)) {
078                                    //noinspection unchecked
079                                    result.add(contributor);
080                            }
081                    }
082                    return result;
083            }
084    
085            /**
086             * {@inheritDoc}
087             */
088            public <T> List<Decorator<T>> getDecorators(final Service<T> service) {
089                    final List<Decorator<T>> result = CollectionUtils.newList();
090                    for (Decorator decorator : decorators) {
091                            if (decorator.supports(service)) {
092                                    //noinspection unchecked
093                                    result.add(decorator);
094                            }
095                    }
096                    return result;
097            }
098    
099            /**
100             * {@inheritDoc}
101             */
102            public <T> List<Interceptor<T>> getInterceptors(final Service<T> service) {
103                    final List<Interceptor<T>> result = CollectionUtils.newList();
104                    for (Interceptor interceptor : interceptors) {
105                            if (interceptor.supports(service)) {
106                                    //noinspection unchecked
107                                    result.add(interceptor);
108                            }
109                    }
110                    return result;
111            }
112    }
