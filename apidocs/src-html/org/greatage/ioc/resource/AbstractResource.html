005    package org.greatage.ioc.resource;
006    
007    import org.greatage.util.I18nUtils;
008    import org.greatage.util.StringUtils;
009    
010    import java.io.BufferedInputStream;
011    import java.io.IOException;
012    import java.io.InputStream;
013    import java.net.URL;
014    import java.util.List;
015    import java.util.Locale;
016    
017    /**
018     * This class represents abstract {@link Resource} implementation that implements all base logic.
019     *
020     * @author Ivan Khalopik
021     * @since 1.0
022     */
023    public abstract class AbstractResource implements Resource {
024            private static final char FILE_DELIMITER = '.';
025            private static final char LOCALE_DELIMITER = '_';
026            private static final char FOLDER_DELIMITER = '/';
027    
028            private static final String CURRENT_FOLDER = ".";
029            private static final String PARENT_FOLDER = "..";
030    
031            private final AbstractResource parent;
032            private final String name;
033            private final Locale locale;
034    
035            private String path;
036    
037            /**
038             * Creates new resource instance with defined parent resource, name and locale.
039             *
040             * @param parent parent resource
041             * @param name   resource name
042             * @param locale resource locale
043             */
044            protected AbstractResource(final AbstractResource parent, final String name, final Locale locale) {
045                    this.parent = parent;
046                    this.name = name;
047                    this.locale = locale;
048            }
049    
050            /**
051             * {@inheritDoc}
052             */
053            public String getName() {
054                    return name;
055            }
056    
057            /**
058             * {@inheritDoc}
059             */
060            public Resource getParent() {
061                    return parent;
062            }
063    
064            /**
065             * {@inheritDoc}
066             */
067            public Locale getLocale() {
068                    return locale;
069            }
070    
071            /**
072             * {@inheritDoc}
073             */
074            public Resource inLocale(final Locale resourceLocale) {
075                    final List<Locale> locales = I18nUtils.getCandidateLocales(resourceLocale);
076                    for (Locale candidate : locales) {
077                            if (candidate.equals(locale) && exists()) {
078                                    return this;
079                            }
080                            final Resource resource = createResource(parent, name, candidate);
081                            if (resource.exists()) {
082                                    return resource;
083                            }
084                    }
085                    return null;
086            }
087    
088            /**
089             * {@inheritDoc}
090             */
091            public Resource getChild(final String resourcePath) {
092                    if (StringUtils.isEmpty(resourcePath) || CURRENT_FOLDER.equals(resourcePath)) {
093                            return this;
094                    }
095                    if (PARENT_FOLDER.equals(resourcePath)) {
096                            return parent;
097                    }
098    
099                    final int slashIndex = resourcePath.indexOf(FOLDER_DELIMITER);
100                    if (slashIndex == -1) {
101                            return createResource(name.lastIndexOf(FILE_DELIMITER) < 0 ? this : parent, resourcePath, locale);
102                    } else {
103                            final Resource child = getChild(resourcePath.substring(0, slashIndex));
104                            return child.getChild(resourcePath.substring(slashIndex + 1));
105                    }
106            }
107    
108            /**
109             * {@inheritDoc}
110             */
111            public boolean exists() {
112                    return toURL() != null;
113            }
114    
115            /**
116             * {@inheritDoc}
117             */
118            public InputStream open() throws IOException {
119                    final URL url = toURL();
120                    if (url == null) {
121                            return null;
122                    }
123                    return new BufferedInputStream(url.openStream());
124            }
125    
126            /**
127             * Gets absolute resource path inside the file system.
128             *
129             * @return absolute resource path, not null
130             */
131            protected String getPath() {
132                    if (path == null) {
133                            final StringBuilder builder = new StringBuilder();
134    
135                            final String folder = parent != null ? parent.getPath() : null;
136                            if (!StringUtils.isEmpty(folder)) {
137                                    builder.append(folder).append(FOLDER_DELIMITER);
138                            }
139    
140                            final int dotPosition = name.lastIndexOf(FILE_DELIMITER);
141                            final String file = dotPosition != -1 ? name.substring(0, dotPosition) : name;
142                            final String extension = dotPosition != -1 ? name.substring(dotPosition) : null;
143                            builder.append(file);
144    
145                            if (locale != null && !StringUtils.isEmpty(locale.toString())) {
146                                    builder.append(LOCALE_DELIMITER).append(locale);
147                            }
148    
149                            if (!StringUtils.isEmpty(extension)) {
150                                    builder.append(extension);
151                            }
152                            path = builder.toString();
153                    }
154                    return path;
155            }
156    
157            /**
158             * Gets URL representation of resource.
159             *
160             * @return resource to URL representation or null
161             */
162            protected abstract URL toURL();
163    
164            /**
165             * Creates new resource instance of needed implementation with defined parent resource, name and locale.
166             *
167             * @param parentResource parent resource
168             * @param resourceName   resource name
169             * @param resourceLocale resource locale
170             * @return new resource instance of needed implementation
171             */
172            protected abstract Resource createResource(final Resource parentResource,
173                                                                                               final String resourceName,
174                                                                                               final Locale resourceLocale);
175    }
