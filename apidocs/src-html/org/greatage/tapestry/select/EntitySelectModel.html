005    package org.greatage.tapestry.select;
006    
007    import org.greatage.domain.Entity;
008    import org.greatage.domain.EntityService;
009    
010    import java.io.Serializable;
011    import java.util.List;
012    
013    /**
014     * Needs ga:ga-core dependency
015     *
016     * @author Ivan Khalopik
017     */
018    public class EntitySelectModel<PK extends Serializable, E extends Entity<PK>>
019                    extends LazySelectModel<E> {
020    
021            private final EntityService<PK, E> entityService;
022    
023            public EntitySelectModel(EntityService<PK, E> entityService) {
024                    this.entityService = entityService; //todo: add defence support
025            }
026    
027            @Override
028            public List<E> getValues() {
029                    return entityService.getEntities();
030            }
031    }
