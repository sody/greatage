005    package org.greatage.tapestry.commonlib.components;
006    
007    import org.apache.commons.codec.binary.Base64;
008    import org.apache.tapestry5.BindingConstants;
009    import org.apache.tapestry5.MarkupWriter;
010    import org.apache.tapestry5.annotations.Parameter;
011    import org.greatage.tapestry.commonlib.base.components.AbstractComponent;
012    
013    import java.io.UnsupportedEncodingException;
014    
015    /**
016     * @author Ivan Khalopik
017     */
018    public class Image extends AbstractComponent {
019    
020            @Parameter(required = true)
021            private byte[] source;
022    
023            @Parameter(defaultPrefix = BindingConstants.LITERAL, value = "image/jpeg")
024            private String contentType;
025    
026            protected boolean beginRender(MarkupWriter writer) {
027                    if (source != null) {
028                            writer.element("img");
029                            final StringBuilder sb = new StringBuilder("data:");
030                            sb.append(contentType).append(";base64,");
031                            sb.append(serializeData(source));
032                            writer.attributes("src", sb.toString());
033                            writer.end();
034                    }
035                    return false;
036            }
037    
038            private String serializeData(byte[] data) {
039                    try {
040                            final byte[] bytes = Base64.encodeBase64(data);
041                            return new String(bytes, "UTF-8");
042                    } catch (UnsupportedEncodingException e) {
043                            return null;
044                    }
045            }
046    }
