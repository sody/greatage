001    package org.greatage.tapestry.internal;
002    
003    import org.apache.tapestry5.internal.services.StreamableResource;
004    
005    import java.io.IOException;
006    import java.io.InputStream;
007    
008    /**
009     * @author Ivan Khalopik
010     * @since 1.1
011     */
012    public class SvgStreamableResource implements StreamableResource {
013            private final StreamableResource delegate;
014    
015            public SvgStreamableResource(final StreamableResource delegate) {
016                    this.delegate = delegate;
017            }
018    
019            public String getContentType() throws IOException {
020                    return "image/svg+xml";
021            }
022    
023            public int getSize(final boolean compress) throws IOException {
024                    return delegate.getSize(compress);
025            }
026    
027            public InputStream getStream(final boolean compress) throws IOException {
028                    return delegate.getStream(compress);
029            }
030    
031            public long getLastModified() throws IOException {
032                    return delegate.getLastModified();
033            }
034    }
