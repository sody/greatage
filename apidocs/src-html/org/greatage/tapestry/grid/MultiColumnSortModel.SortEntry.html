005    package org.greatage.tapestry.grid;
006    
007    import org.apache.tapestry5.beaneditor.BeanModel;
008    import org.apache.tapestry5.beaneditor.PropertyModel;
009    import org.apache.tapestry5.grid.ColumnSort;
010    import org.apache.tapestry5.grid.GridModel;
011    import org.apache.tapestry5.grid.GridSortModel;
012    import org.apache.tapestry5.grid.SortConstraint;
013    
014    import java.util.ArrayList;
015    import java.util.List;
016    
017    /**
018     * @author Ivan Khalopik
019     */
020    public class MultiColumnSortModel implements GridSortModel {
021            private final List<SortEntry> entries;
022            private final GridModel gridModel;
023            private final int depth;
024    
025            public MultiColumnSortModel(final GridModel gridModel, final List<SortEntry> entries, int depth) {
026                    this.gridModel = gridModel;
027                    this.entries = entries;
028                    this.depth = depth;
029                    verifyDepth();
030            }
031    
032            public ColumnSort getColumnSort(final String columnId) {
033                    final SortEntry sortEntry = getSortEntry(columnId);
034                    return sortEntry == null ? ColumnSort.UNSORTED : sortEntry.isAscending() ? ColumnSort.ASCENDING : ColumnSort.DESCENDING;
035            }
036    
037            public void updateSort(final String columnId) {
038                    final SortEntry sortEntry = getSortEntry(columnId);
039                    if (sortEntry != null) {
040                            sortEntry.toggle();
041                            entries.remove(sortEntry);
042                            entries.add(0, sortEntry);
043                    } else {
044                            entries.add(0, new SortEntry(columnId));
045                    }
046                    verifyDepth();
047            }
048    
049            private void verifyDepth() {
050                    while (entries.size() > depth) {
051                            entries.remove(entries.size() - 1);
052                    }
053            }
054    
055            public List<SortConstraint> getSortConstraints() {
056                    final List<SortConstraint> constraints = new ArrayList<SortConstraint>();
057                    for (SortEntry entry : entries) {
058                            constraints.add(createSortConstraint(entry));
059                    }
060                    return constraints;
061            }
062    
063            public void clear() {
064                    entries.clear();
065            }
066    
067            private SortEntry getSortEntry(final String columnId) {
068                    for (SortEntry entry : entries) {
069                            if (entry.getColumnId().equals(columnId)) {
070                                    return entry;
071                            }
072                    }
073                    return null;
074            }
075    
076            private SortConstraint createSortConstraint(final SortEntry entry) {
077                    final ColumnSort sort = entry.isAscending() ? ColumnSort.ASCENDING : ColumnSort.DESCENDING;
078                    final BeanModel dataModel = gridModel.getDataModel();
079                    PropertyModel propertyModel;
080                    try {
081                            propertyModel = dataModel.getById(entry.getColumnId());
082                    } catch (Exception e) {
083                            //todo: remove this
084                            propertyModel = dataModel.add(entry.getColumnId(), null);
085                    }
086                    return new SortConstraint(propertyModel, sort);
087            }
088    
089            public static class SortEntry {
090                    private final String columnId;
091                    private boolean ascending;
092    
093                    public SortEntry(final String columnId) {
094                            this.columnId = columnId;
095                    }
096    
097                    public String getColumnId() {
098                            return columnId;
099                    }
100    
101                    public boolean isAscending() {
102                            return ascending;
103                    }
104    
105                    public void toggle() {
106                            ascending = !ascending;
107                    }
108            }
109    }
