7   import org.greatage.util.EncodeUtils;
8   import org.greatage.util.StringUtils;

10  /**
11   * @author Ivan Khalopik

14  public class MessageDigestPasswordEncoder implements PasswordEncoder {
15  	private final String algorithm;
16  	private final boolean useBase64;
17  
18  	public MessageDigestPasswordEncoder(final String algorithm, final boolean useBase64) {
19  		this.algorithm = algorithm;
20  		this.useBase64 = useBase64;
21  	}
22  
23  	public String encode(final String password) {
24  		if (password == null) {
25  			return null;
26  		}
27  		final byte[] bytes = EncodeUtils.encode(password.getBytes(), algorithm);
28  		return useBase64 ? new String(EncodeUtils.encodeBase64(bytes)) : StringUtils.toHexString(bytes);
29  	}
30  }
