001    package org.greatage.util;
002    
003    import org.testng.Assert;
004    import org.testng.annotations.DataProvider;
005    import org.testng.annotations.Test;
006    
007    import java.util.List;
008    import java.util.Locale;
009    
010    /**
011     * @author Ivan Khalopik
012     * @since 1.0
013     */
014    public class TestI18nUtils extends Assert {
015    
016            @DataProvider
017            public Object[][] getLocaleData() {
018                    return new Object[][]{
019                                    {null, null},
020                                    {"", I18nUtils.ROOT_LOCALE},
021                                    {"en", Locale.ENGLISH},
022                                    {"en_US", Locale.US},
023                                    {"en_GB", Locale.UK},
024                                    {"ru", new Locale("ru")},
025                                    {"ru_RU_xxx", new Locale("ru", "RU", "xxx")},
026                    };
027            }
028    
029            @Test(dataProvider = "getLocaleData")
030            public void testGetLocale(String locale, Locale expected) {
031                    final Locale actual = I18nUtils.getLocale(locale);
032                    assertEquals(actual, expected);
033            }
034    
035            @DataProvider
036            public Object[][] getCandidateLocalesData() {
037                    return new Object[][]{
038                                    {null, CollectionUtils.newList(I18nUtils.ROOT_LOCALE)},
039                                    {I18nUtils.ROOT_LOCALE, CollectionUtils.newList(I18nUtils.ROOT_LOCALE)},
040                                    {Locale.ENGLISH, CollectionUtils.newList(Locale.ENGLISH, I18nUtils.ROOT_LOCALE)},
041                                    {Locale.US, CollectionUtils.newList(Locale.US, Locale.ENGLISH, I18nUtils.ROOT_LOCALE)},
042                                    {
043                                                    new Locale("en", "US", "xxx"), CollectionUtils
044                                                    .newList(new Locale("en", "US", "xxx"), Locale.US, Locale.ENGLISH, I18nUtils.ROOT_LOCALE)
045                                    },
046                    };
047            }
048    
049            @Test(dataProvider = "getCandidateLocalesData")
050            public void testGetCandidateLocales(Locale locale, List<Locale> expected) {
051                    final List<Locale> actual = I18nUtils.getCandidateLocales(locale);
052                    assertEquals(actual, expected);
053            }
054    }
