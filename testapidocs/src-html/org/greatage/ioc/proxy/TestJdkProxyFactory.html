024            @DataProvider
025            public Object[][] createProxyData() {
026                    return new Object[][]{
027                                    {

029                                                    CollectionUtils.newList(),
030                                                    "test", "test1"
031                                    },
032                                    {
033                                                    new MockObjectBuilder<MockIOCInterface>(MockIOCInterface.class, MockIOCInterfaceImpl2.class, "test2"),
034                                                    null,
035                                                    "test", "test2"
036                                    },
037                                    {
038                                                    new MockObjectBuilder<MockIOCInterface>(MockIOCInterface.class, MockIOCInterfaceImpl2.class),
039                                                    null,
040                                                    "test", null
041                                    },
042                                    {
043                                                    new MockObjectBuilder<MockIOCInterface>(MockIOCInterface.class, MockIOCInterfaceImpl3.class),
044                                                    CollectionUtils.newList(
045                                                                    new MethodAdvice() {
046                                                                            public Object advice(final Invocation invocation, final Object... parameters) throws Throwable {
047                                                                                    return "advice:" + invocation.proceed(parameters);
048                                                                            }
049                                                                    }
050                                                    ),
051                                                    "advice:test", "advice:" + MockIOCInterfaceImpl3.MESSAGE
052                                    },
053                                    {
054                                                    new MockObjectBuilder<MockIOCInterface>(MockIOCInterface.class, MockIOCInterfaceImpl4.class, new MockIOCInterfaceImpl3()),
055                                                    CollectionUtils.newList(
056                                                                    new MethodAdvice() {
057                                                                            public Object advice(final Invocation invocation, final Object... parameters) throws Throwable {
058                                                                                    return "advice1:" + invocation.proceed(parameters);
059                                                                            }
060                                                                    },
061                                                                    new MethodAdvice() {
062                                                                            public Object advice(final Invocation invocation, final Object... parameters) throws Throwable {
063                                                                                    return "advice2:" + invocation.proceed(parameters);

065                                                                    }
066                                                    ),
067                                                    "advice2:advice1:test", "advice2:advice1:" + MockIOCInterfaceImpl3.MESSAGE
068                                    },
069                    };

074                    return new Object[][]{

096            }
097    
098            @BeforeClass
099            public void setupProxyFactory() {
100                    proxyFactory = new JdkProxyFactory();
101            }
102    
103            @Test(dataProvider = "createProxyData")
104            public void testCreateProxy(final ObjectBuilder<? extends MockIOCInterface> builder,
105                                                                    final List<MethodAdvice> advices,
106                                                                    final String expected1,
107                                                                    final String expected2) {
108                    final MockIOCInterface proxy = proxyFactory.createProxy(builder, advices);
109                    Assert.assertEquals(proxy.say("test"), expected1);
110                    Assert.assertEquals(proxy.say(), expected2);
111            }
112    
113            @Test(dataProvider = "createProxyWrongData", expectedExceptions = IllegalArgumentException.class)
114            public void testCreateProxyWrong(final ObjectBuilder<? extends MockIOCInterface> builder,
115                                                                             final List<MethodAdvice> advices) {
116                    final MockIOCInterface proxy = proxyFactory.createProxy(builder, advices);
117                    proxy.say("test");
118                    proxy.say();
119            }
